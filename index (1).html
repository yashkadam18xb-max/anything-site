import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.model_selection import train_test_split
from statsmodels.stats.weightstats import ztest

# Load the dataset
df = pd.read_csv('insurance.csv')

print("="*50)
print("DATASET LOADED")
print("="*50)
print(df.head())
print(f"\nTotal Records: {len(df)}")

# -----------------------------------------------
# A. Partition the dataset (75% Train, 25% Test)
# -----------------------------------------------
print("\n" + "="*50)
print("A. PARTITIONING DATASET (75% / 25%)")
print("="*50)

train_df, test_df = train_test_split(df, test_size=0.25, random_state=42)

# -----------------------------------------------
# B. Identify total records in each set
# -----------------------------------------------
print("\n" + "="*50)
print("B. RECORD COUNT")
print("="*50)

train_count = len(train_df)
test_count  = len(test_df)

print(f"Total number of records in the Training Data Set: {train_count}")
print(f"Total number of records in the Testing Data Set : {test_count}")

# Bar Graph to confirm proportions
labels   = ['Training Set', 'Testing Set']
counts   = [train_count, test_count]
colors   = ['steelblue', 'lightcoral']
percents = [train_count/len(df)*100, test_count/len(df)*100]

plt.figure(figsize=(8, 6))
bars = plt.bar(labels, counts, color=colors, edgecolor='black', width=0.4)

for bar, count, pct in zip(bars, counts, percents):
    plt.text(bar.get_x() + bar.get_width()/2,
             bar.get_height() + 10,
             f'{count} ({pct:.1f}%)',
             ha='center', va='bottom', fontweight='bold', fontsize=12)

plt.title('Distribution of Records: Training vs Testing Set', fontsize=14)
plt.ylabel('Number of Records')
plt.ylim(0, max(counts) + 150)
plt.tight_layout()
plt.show()

# -----------------------------------------------
# C. Validate partition using Two-Sample Z-Test
# -----------------------------------------------
print("\n" + "="*50)
print("C. TWO-SAMPLE Z-TEST VALIDATION (on 'bmi' column)")
print("="*50)

z_stat, p_val = ztest(train_df['bmi'], test_df['bmi'])

print(f"\n--- Two-Sample Z-Test Validation (on 'bmi' column) ---")
print(f"Z-statistic : {z_stat:.4f}")
print(f"P-value     : {p_val:.4f}")

if p_val > 0.05:
    print("\nConclusion: Fail to reject the null hypothesis.")
    print("The training and test sets are statistically similar.")
else:
    print("\nConclusion: Reject the null hypothesis.")
    print("There is a significant difference between the partitions.")

print("\n" + "="*50)
print("CONCLUSION: Data Modelling using Python Complete!")
print("="*50)
Experiment 04
Statistical Hypothesis Testing using Scipy
Shapiro-Wilk, D'Agostino K², and Anderson-Darling normality tests on BMI column.
Run on: Google Colab · scipy is pre-installed · No extra installs needed
Full Code
import pandas as pd
from scipy.stats import shapiro, normaltest, anderson

# Load the dataset
df = pd.read_csv('insurance.csv')
data = df['bmi']

print("="*50)
print("NORMALITY TESTING FOR BMI")
print("="*50)

# -----------------------------------------------
# A.1 Shapiro-Wilk Test
# -----------------------------------------------
print("\n" + "="*50)
print("1. SHAPIRO-WILK TEST")
print("="*50)

stat_sw, p_sw = shapiro(data)
print(f"Shapiro-Wilk Test: Statistics={stat_sw:.3f}, p-value={p_sw:.3f}")

if p_sw > 0.05:
    print("Result: Sample looks Gaussian (fail to reject H0)")
else:
    print("Result: Sample does not look Gaussian (reject H0)")

# -----------------------------------------------
# A.2 D'Agostino's K^2 Test
# -----------------------------------------------
print("\n" + "="*50)
print("2. D'AGOSTINO'S K^2 TEST")
print("="*50)

stat_k2, p_k2 = normaltest(data)
print(f"D'Agostino's K^2 Test: Statistics={stat_k2:.3f}, p-value={p_k2:.3f}")

if p_k2 > 0.05:
    print("Result: Sample looks Gaussian (fail to reject H0)")
else:
    print("Result: Sample does not look Gaussian (reject H0)")

# -----------------------------------------------
# A.3 Anderson-Darling Test
# -----------------------------------------------
print("\n" + "="*50)
print("3. ANDERSON-DARLING TEST")
print("="*50)

result_ad = anderson(data)
print(f"Anderson-Darling Test: Statistic={result_ad.statistic:.3f}")

for i in range(len(result_ad.critical_values)):
    sl, cv = result_ad.significance_level[i], result_ad.critical_values[i]
    if result_ad.statistic < cv:
        print(f"Result at {sl}% level: Sample looks Gaussian (Statistic < {cv})")
    else:
        print(f"Result at {sl}% level: Sample does not look Gaussian (Statistic > {cv})")

print("\n" + "="*50)
print("CONCLUSION: Statistical Hypothesis Test Complete!")
print("="*50)

#    Features: Age and Charges

print("="*50)

df['Cluster'] = kmeans.fit_predict(X_scaled)

print("\nK-Means Clustering Done! (K=3)")
print("\nFirst 5 records with assigned clusters:")
print(df[['age', 'charges', 'Cluster']].head())

import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler
from sklearn.neighbors import NearestNeighbors

# Load the dataset
df = pd.read_csv('insurance.csv')

print("="*50)
print("DATASET LOADED")
print("="*50)
print(df.head())
print(f"\nShape: {df.shape}")

# -----------------------------------------------
# Preprocessing
# Convert categorical columns to numbers
# -----------------------------------------------
df['is_smoker'] = df['smoker'].map({'yes': 1, 'no': 0})
df['is_female'] = df['sex'].map({'female': 1, 'male': 0})

# Features used for finding similar profiles
features = ['age', 'bmi', 'is_smoker', 'children']
X = df[features]

# Scale the features
scaler   = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Build KNN model (K=5, Euclidean distance)
model = NearestNeighbors(n_neighbors=5, metric='euclidean')
model.fit(X_scaled)

print("\nKNN Model Trained Successfully!")
print(f"Features used : {features}")
print(f"Neighbours    : 5")
print(f"Metric        : Euclidean Distance")

# -----------------------------------------------
# Recommendation Function
# -----------------------------------------------
def recommend_insurance_profile(age, bmi, smoker_status, children):

    print("\n" + "="*50)
    print("INSURANCE RECOMMENDATION SYSTEM")
    print("="*50)

    is_smoker      = 1 if smoker_status.lower() == 'yes' else 0
    new_user       = pd.DataFrame(
                        [[age, bmi, is_smoker, children]],
                        columns=features
                     )
    new_user_scaled = scaler.transform(new_user)

    distances, indices = model.kneighbors(new_user_scaled)
    similar_people     = df.iloc[indices[0]]
    avg_charges        = similar_people['charges'].mean()

    print(f"\n--- Recommendations for Profile ---")
    print(f"Age          : {age}")
    print(f"BMI          : {bmi}")
    print(f"Smoker       : {smoker_status}")
    print(f"Children     : {children}")

    print(f"\nTop 5 Similar Profiles Found:")
    print(similar_people[['age', 'bmi', 'smoker',
                           'children', 'charges']].to_string())

    print(f"\nSuggested Insurance Budget : ${avg_charges:.2f}")

    if avg_charges > 30000:
        plan = "Platinum Comprehensive Plan (High Coverage)"
    elif avg_charges > 15000:
        plan = "Gold Standard Plan (Moderate Coverage)"
    else:
        plan = "Silver Essential Plan (Basic Coverage)"

    print(f"Recommended Plan           : {plan}")
    print("\nBased on 5 similar profiles found in our database.")
    print("="*50)

# -----------------------------------------------
# Test the Recommendation System
# -----------------------------------------------
print("\n--- TEST 1 ---")
recommend_insurance_profile(
    age=19, bmi=27.9, smoker_status='yes', children=0
)

print("\n--- TEST 2 ---")
recommend_insurance_profile(
    age=28, bmi=33.0, smoker_status='yes', children=0
)

print("\n--- TEST 3 ---")
recommend_insurance_profile(
    age=45, bmi=22.0, smoker_status='no', children=2
)

print("\n" + "="*50)
print("CONCLUSION: Recommendation System Complete!")
print("="*50)

    schema=schema
)

print(f"\nTotal Records Loaded: {batch_df.count()}")
print("\nFirst 5 rows:")
batch_df.show(5)

# Batch Analysis 1: Avg charges by smoker status
print("\n--- Average Charges by Smoker Status ---")
batch_avg = batch_df.groupBy("smoker") \
                    .agg(avg("charges"))
batch_avg.show()

# Batch Analysis 2: Avg charges by region
print("\n--- Average Charges by Region ---")
batch_region = batch_df.groupBy("region") \
                       .agg(avg("charges").alias("avg_charges")) \
                       .orderBy("avg_charges", ascending=False)
batch_region.show()

# Batch Analysis 3: Convert to Pandas for summary
print("\n--- Batch Summary in Pandas Format ---")
pandas_batch = batch_region.toPandas()
print(pandas_batch)
